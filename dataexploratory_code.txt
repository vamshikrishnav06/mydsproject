def uni_biv(data,var):
    if(data[var[0]].dtype == np.float64 or data[var[0]].dtype == np.int64):
        if len(var)==1:
            data[var].hist()
            um=plt.plot()
        else:
            um=data.plot.scatter(x=var[0],y=var[1])
    else:
        var1= ''.join(map(str,var))
        um=data[var1].value_counts().sort_index().plot.bar()
    return um
